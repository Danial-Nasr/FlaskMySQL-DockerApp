CREATE DATABASE irs_data;

USE irs_data;

CREATE TABLE taxpayers (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    age INT,
    state VARCHAR(50)
);

CREATE TABLE returns (
    id INT AUTO_INCREMENT PRIMARY KEY,
    taxpayer_id INT,
    year INT,
    tax_paid DECIMAL(10, 2),
    refund DECIMAL(10, 2),
    filing_type VARCHAR(50),
    FOREIGN KEY (taxpayer_id) REFERENCES taxpayers(id)
);
